24  Program on side B to find a coincidence time difference. Takes a directory
25  or file with type-2 and type-1 files, and concatenates a given number of
26  packets starting with a given epoch number, evaluates the time difference
27  in multiples of 1/8 nsec, and returns that to stdout or any other file,
28  eventually with some reliability information on the found timing difference.
29 
30 
31  usage: pfind [-i type-2 streamfile] | [-d type-2 directory] | [-s socket2]
32  [-I type-1 streamfile] | [-D type-1 directory] | [-S socket1]
33  -e startepoch
34  [-n epochnums]
35  [-l logfilename] [-v verbosity]
36  [-k] [-K]
37  [-r resolution]
38  [-q bufferwidth ]
39 
40  DATA STREAM OPTIONS:
41  -i infile2: filename of type-2 packets. Can be a file or a socket
42  and has to supply binary data according to the type-2
43  data spec from the chopper program.
44  -d dir2: All type-2 packets are saved into the directory dir2, with
45  the file name being the epoch (filling zero expanded)
46  in hex. Filename is not padded at end.
47  -I infile1: filename of type-1 packets. Can be a file or a socket
48  and has to supply binary data according to the type-1
49  data spec from the chopper2 program.
50  -D dir1: All type-1 packets are saved into the directory dir1, with
51  the file name being the epoch (filling zero expanded)
52  in hex. Filename is not padded at end.
53  -k : if set, type-2 streams are removed aafter consumption
54  if the directory input has been chosen.
55  -K : if set, type-1 streams are removed aafter consumption
56  if the directory input has been chosen.
57 
58  DATA MANAGEMENT OPTIONS:
59  -e startepoch epoch to start with. default is 0.
60  -n epochnums define a runtime of epochums epochs before looking for a
61  time delay. default is 1.

63  RESOLUTION:

65  to closest power of 1 nsec. Default is 2 nsec.
66  -q bufferwidth order of FFT buffer size. Defines the wraparound size

220 /* function to fill buffer with stream-1 raw data. eats an input bufferpointer,
221  a file handle, a max size in bytes, and a pointer to a header_1 struct.
222  returns an error code. */
223 int get_stream_1(void *buffer, int handle, int maxsize, struct header_1 *head) {
224  int retval;
225  int eidx;
226  unsigned int *ib = buffer;
227  struct header_1 *h; /* for local storage */
228  retval = read(handle, buffer, maxsize);
229  if (!retval) return 12; /* nothing available */
230  if (!(retval + 1)) return 13; /* other error */
231  if (retval < (int)sizeof(struct header_1)) return 14; /* incomplete read */
232  h = (struct header_1 *)buffer; /* at beginning of stream */
233  /* consistency check at end */
234  if ((h->tag != TYPE_1_TAG) && (h->tag != TYPE_1_TAG_U)) return 27;
235  if (h->length) {
236  eidx = (h->length * sizeof(struct rawevent) + sizeof(struct header_1)) /
237  sizeof(unsigned int);
238  if (eidx != (retval / (int)sizeof(unsigned int) - 2))
239  return 15; /* length mismatch */
240  if (ib[eidx] | ib[eidx + 1]) return 15; /* last word nonzero */
241 
242  } else {
243  if (retval - sizeof(struct header_1) % sizeof(struct rawevent))
244  return 15; /* size mismatch */
245  eidx = retval / sizeof(unsigned int);
246  if (ib[eidx - 1] | ib[eidx - 2]) return 15; /* last word nonzero */
247  if (!(ib[eidx - 3] | ib[eidx - 4])) return 15; /* last real entry zero */
248  h->length =
249  (retval - sizeof(struct header_1)) / sizeof(struct rawevent) - 1;
250  }
251  *head = h[0]; /* return pointer to header */
252  return 0;
253 }

295 /* fill complex field with int filed, do transform and retrun max/sigma */
296 void findmax(int *buf1, int *buf2, double *maxval, double *sigma, double *mean,
297  int *pos, int size, int ecnt1, int ecnt2) {
298  int i; /* counter index */
299  double ai, ar, br, bi; /* temporary variables */
300  double sx, sxx;
301  int maxpos;
302 
303  /* clear double buffers / transfer int to double buffers */
304  ar = ((double)ecnt1) / size; /* to get mean at zero */
305  br = ((double)ecnt2) / size;
306  for (i = 0; i < size; i++) {
307  f1[i][0] = (double)buf1[i] - ar;
308  f1[i][1] = 0.;
309  f2[i][0] = (double)buf2[i] - br;
310  f2[i][1] = 0.;
311  }
312  /* do forward transformations */
313  fftw_execute(plan1);
314  fftw_execute(plan2);
315  /* do conjugate and multiplication into array 1 */
316  for (i = 0; i < size; i++) {
317  ar = f1[i][0];
318  ai = f1[i][1];
319  br = f2[i][0];
320  bi = f2[i][1];
321  f1[i][0] = ar * br + ai * bi;
322  f1[i][1] = ar * bi - ai * br;
323  }
324  /* do do backtransform */
325  fftw_execute(plan3);
326  /* evaluate max, stddev and mean */
327  *maxval = 0.;
328  maxpos = 0;
329  sxx = 0;
330  sx = 0.;
331  for (i = 0; i < size; i++) {
332  ar = f1[i][0];
333  if (ar > *maxval) {
334  maxpos = i;
335  *maxval = ar;
336  }
337  sx += ar;
338  sxx += (ar * ar);
339  }
340  /* return values properly back */
341  *mean = sx / size;
342  *sigma = sqrt(sxx / size - (*mean) * (*mean));
343  *pos = maxpos;
344 }
