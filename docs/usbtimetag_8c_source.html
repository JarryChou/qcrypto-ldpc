248 /* routine to allocate DMA buffer RAM of size (in bytes).
249  Returns 0 on success, and <0 on fail */
250 static int get_dma_buffer(ssize_t size, struct cardinfo *cp) {
251  ssize_t bytes_to_get = size & ~0x3; /* get long-aligned */
252  ssize_t usedbytes;
253  unsigned long page_order;
254  void * bufferpiece;
255  dma_addr_t physicaladdress;
256 
257  struct dma_page_pointer * currbuf;
258  struct dma_page_pointer * tmpbuf;
259  /* check multi pages */
260  struct page *spa;
261  unsigned long pflags;
262  int pcnt,i;
263 
264  /* reset dma pointer buffer */
265  currbuf=NULL; /* dma_main_pointer; */ /* NULL if no buffer exists */
266 
267  /* still have to get only small pieces....?? */
268  page_order = 4;
269 
270  /* page_order = get_order(bytes_to_get); */
271  if (page_order >= MAX_ORDER) page_order=MAX_ORDER;
272 
273  while (bytes_to_get>0) {
274  /* shrink size if possible */
275  while((page_order>0) && (PAGE_SIZE<<(page_order-1))>=bytes_to_get)
276  page_order--;
277 
278  /* transition to dma kernel tools */
279  bufferpiece = dma_alloc_coherent(cp->hostdev,
280  PAGE_SIZE<<page_order,
281  &physicaladdress,
282  GFP_KERNEL);
283  /* old style: */
284  /* bufferpiece = (void *)__get_free_pages(GFP_KERNEL,page_order); */
285 
286  if (bufferpiece) {
287 
288  /* repair missing page counts */
289  add_individual_page_counts(bufferpiece, page_order);
290 
291  /* get block structure */
292  for (i=0;i<(1 << page_order);i++) {
293  spa=virt_to_page(bufferpiece+i*PAGE_SIZE);
294  pcnt=page_count(spa);
295  pflags = spa->flags;
296  }
297  /* success: make new entry in chain */
298  tmpbuf = (dma_page_pointer *) kmalloc(sizeof(dma_page_pointer),
299  GFP_KERNEL);
300  if (!tmpbuf) {
301  printk(" Wruagh - kmalloc failed for buffer pointer....\n");
302  /* give it back */
303  /* old style:
304  free_pages((unsigned long)bufferpiece,page_order); */
305  dma_free_coherent(cp->dev->bus->controller, /* get a device */
306  PAGE_SIZE<<page_order,
307  bufferpiece, physicaladdress);
308  printk("kmalloc failed during DMA buffer alloc.\n");
309  return -ENOMEM;
310  }
311 
312 
313  if (currbuf) { /* there is already a structure */
314  /* fill new struct; currbuf points to last structure filled */
315  tmpbuf->next=currbuf->next; tmpbuf->previous=currbuf;
316  /* insert in chain */
317  currbuf->next->previous=tmpbuf;currbuf->next=tmpbuf;
318  currbuf=tmpbuf;
319  } else {
320  tmpbuf->previous=tmpbuf;
321  tmpbuf->next=tmpbuf; /* fill new struct */
322  currbuf=tmpbuf;
323  cp->dma_main_pointer=currbuf; /* set main pointer */
324  };
325 
326  /* fill structure with actual buffer info */
327  usedbytes = PAGE_SIZE<<page_order;
328  currbuf->fullsize = usedbytes; /* all allocated bytes */
329  usedbytes = (usedbytes>bytes_to_get?bytes_to_get:usedbytes);
330  currbuf->size=usedbytes; /* get useful size into buffer */
331  currbuf->order=page_order; /* needed for free_pages */
332  currbuf->buffer=bufferpiece; /* kernel address of buffer */
333  currbuf->physicaladdress=physicaladdress; /* used for DMA later */
334 
335  /* less work to do.. */
336  bytes_to_get -= usedbytes;
337  } else {
338  /* could not get the large mem piece. try smaller ones */
339  if (page_order>0) {
340  page_order--; continue;
341  } else {
342  break; /* stop and clean up in case of problems */
343  };
344  }
345  }
346  if (bytes_to_get <=0) {
347  cp->smallpageorder = page_order;
348  cp->minmempiece = (4096 << page_order);
349  return 0; /* everything went fine.... */
350  }
351  /* cleanup of unused buffers and pointers with standard release code */
352  release_dma_buffer(cp);
353  return -ENOMEM;
354 }

368 /* completion handler for urbs; this callback should re-populate urbs as
369  they fall free */
370 /* old code seemed to use other prototype; no idea when this went, it was
371  already away in kernel version 2.6.22. regs not needed, so just drop it
372  and hope I get a feedback if itbreaks something old:
373  static void completion_handler(struct urb *urb, struct pt_regs *regs) { */
374 static void completion_handler(struct urb *urb) {
375  struct cardinfo *cp=(struct cardinfo *)urb->context;
376  unsigned int jf,jd; /* stores jiffies and difference */
377  int tfl; /* transfer length for the next round */
378  /* test about the status */
379  if (urb->status) { /* something happened */
380  cp->transfers_running=0;
381  printk("urb accident; status: %d\n",urb->status);
382  cp->errstat=urb->status;
383  } else { /* urb is finished */
384  /* fix current_received count and clean up mem if necessary */
385  if (urb->actual_length < urb->transfer_buffer_length) {
386  /* patch up rest with zeros; this needs transfer_buffer to
387  contain the virtual address of the DMA buffer; I hope
388  that the urb handlers don't spoil this and allow me
389  to use that unused variable quietly.... */
390  memset(&((char *)urb->transfer_buffer)[urb->actual_length],
391  0, urb->transfer_buffer_length-urb->actual_length);
392  }
393  /* notify reader */
394  cp->received_bytes += urb->transfer_buffer_length;
395 
396  /* buffer length servo to keep interrupt rate below 100 Hz */
397  jf=jiffies; jd=(jf-cp->oldjiffies)*256; cp->oldjiffies = jf;
398  cp->avgdiff += (((int)jd)-cp->avgdiff)/8; /* lowpass over 8 irq calls */
399  /* This is some upper limit to avoid a latch-up of the servo */
400  if (cp->avgdiff > 0x10000) cp->avgdiff=0x10000;
401  if ((cp->jiffservocounter--)<=0) { /* now we can consider changing */
402  cp->jiffservocounter=DEFAULT_JIFFSERVOPERIODE; /* reload counter */
403  if (cp->avgdiff <256) {/* less than 1 jiffie difference */
404  /* increase periode if possible */
405  if (cp->current_transferlength < cp->minmempiece) {
406  cp->current_transferlength <<=1;
407  printk("%s: transfer len increased to %d; avgdiff: %d, jd: %d\n",
408  USBDEV_NAME,
409  cp->current_transferlength,cp->avgdiff,jd);
410  }
411  }
412  if (cp->avgdiff >2500) {/* more than 10 jiffies difference */
413  /* decrease periode if possible */
414  if (cp->current_transferlength > cp->maxpacket) {
415  cp->current_transferlength >>=1;
416  printk("%s: transfer len decreased to %d; avgdiff: %d, jd:%d\n",
417  USBDEV_NAME,
418  cp->current_transferlength,cp->avgdiff,jd);
419  }
420  }
421 
422  }
423 
424  if (cp->transfers_running) { /* we still can submit urbs... */
425  /* submit new URB */
426  urb->transfer_flags = URB_NO_TRANSFER_DMA_MAP;
427  urb->transfer_dma = cp->current_free_mempiece->physicaladdress +
428  cp->current_free_offset;
429  /* for possibly zeroing afterwards */
430  urb->transfer_buffer =
431  &cp->current_free_mempiece->buffer[cp->current_free_offset];
432  /* make sure we never exceed a mem page boundary with transfer */
433  tfl=cp->current_transferlength;
434  if (cp->current_free_offset + tfl >
435  cp->current_free_mempiece->size)
436  tfl=(cp->current_free_mempiece->size)-cp->current_free_offset;
437 
438  /* hopefully this complies with the system to read larger
439  quantities for IN transfers */
440  urb->transfer_buffer_length = tfl;
441  usb_submit_urb(urb,GFP_ATOMIC); /* this is not irq context */
442 
443  /* get prepare next free address */
444  cp->current_free_offset += tfl;
445  if (cp->current_free_offset >= cp->current_free_mempiece->size) {
446  /* we exceeded this page thing */
447  cp->current_free_mempiece = cp->current_free_mempiece->next;
448  cp->current_free_offset = 0;
449  }
450  }
451  }
452 }

706 /* change in the ioctl structure to unlocked_ioctl...removed inode parameter */
707 static long usbdev_flat_ioctl(struct file *filp, unsigned int cmd, unsigned long arg) {
708  struct cardinfo *cp = (struct cardinfo *)filp->private_data;
709  unsigned char *data = cp->scratchbuf; /* DMA-compat buffer */
710  unsigned char len=3; unsigned char chksum=0;
711  int err;
712  int atrf; /* actually transferred data */
713 
714  if (!cp->dev) return -ENODEV;
715 
716  switch (cmd) { /* to distill out real command */
717  /* simple commands which send direct control urbs to the device */
718  case SendDac: case Send_RF_parameter: /* 16 bit argument */
719  case SetWarningwatermark:
720  data[3]=(arg>>8) & 0xff; /* MSbyte */
721  len++; chksum+=data[3];
722  case Rf_Reference: case Autoflush: /* one-byte parameter urbs */
723  case RequestStatus: case WriteBootEEPROM:
724  data[2]=arg & 0xff; /* LSbyte */
725  len++; chksum+=data[2];
726  case Reset_Timestampcard: case InitDac: case InitializeRFSRC:
727  case Set_Inhibitline: case Reset_Inhibitline: case Set_calibration:
728  case Clear_Calibration: case Initialize_FIFO: case Stop_nicely:
729  case StartTransfer: case FreshRestart: case SlowCounterReset:
730  case PartialFIFOreset:
731  data[0]=len; chksum +=len;
732  data[1]=cmd & 0xff; chksum +=data[1];
733  data[len-1]=chksum;
734  /* just send the last significant byte to the device */
735  /* wait for 1 sec */
736  err=usb_bulk_msg(cp->dev, cp->outpipe1, data, len, &atrf, 100);
737  return err?err:atrf; /* return error number of transferred bytes */
738  break;
739 
740  /* here start the commands which deal with the local dma engine */
741  case Start_USB_machine:
742  /* first check if we have the mmap done already */
743  if (!cp->dma_main_pointer || cp->transfers_running)
744  return -EBUSY;
745  /* Ok, we can start. Let's populate the urbs first....*/
746  cp->transfers_running=1;
747  initial_fillurbqueue(cp);
748  break;
749  case Stop_USB_machine: /* A rather brutal kill for now */
750  if (!cp->dma_main_pointer || !cp->transfers_running)
751  return -EBUSY; /* nothing to do or in wrong state */
752  shutdown_urbs(cp);
753  break;
754  case Get_transferredbytes:
755  /* does not make sense if not running */
756  if ((!cp->dma_main_pointer) || (!cp->transfers_running))
757  return -EBUSY;
758  /* otherwise get da value...modulo 0x80000000 to reserve
759  the negative sign to error conditions. ATTENTION: The
760  application must check if an overrun can take place!!!!
761  */
762  return already_transferred_bytes(cp);
763  break;
764 
765  case Reset_Buffering:
766  break;
767  case Get_errstat:
768  return cp->errstat;
769  break;
770  default:
771  return -ENOSYS; /* function not implemented */
772  }
773  return 0; /* went ok... */
774 }
