116 int initDataStruct(ProcessBlock* processBlock) {
117  if (processBlock->algorithmDataPtr) {
118  // the data ptr should be null or else a logic err occurred
119  return 84;
120  }
121  switch (processBlock->algorithmDataMngr->DATA_TYPE) {
122  case ALG_DATATYPE_QBER:
123  processBlock->algorithmDataPtr = malloc2(sizeof(QberData));
124  break;
125  case ALG_DATATYPE_CASCADE:
126  processBlock->algorithmDataPtr = malloc2(sizeof(CascadeData));
127  break;
128  default:
129  fprintf(stderr, "Unsupported data type %d @ initDataStruct\n",
130  processBlock->algorithmDataMngr->DATA_TYPE);
131  return 81;
132  }
133  /* if malloc failed */
134  if (!(processBlock->algorithmDataPtr))
135  return 34;
136  return 0;
137 }
