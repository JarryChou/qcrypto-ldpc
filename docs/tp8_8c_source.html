441 static void ctrlGetStatus() {
442  unsigned char a;
443  SUDPTRCTL=1; /* simple send...just don't use SUDPTRL */
444  EP0BCH = 0x00; EP0BUF[1] = 0x00; /* same for all requests */
445  switch (SETUPDAT[0]) { /* bmRequest */
446  case 0x80: // IN, Device (Remote Wakeup and Self Powered Bit)
447  EP0BUF[0] = 0x01; /* no Remote Wakeup, Self-powerd Device */
448  EP0BCL = 0x02; /* 2 bytes, triggers transfer */
449  break;
450 
451  case 0x81: // IN, Get Status/Interface
452  EP0BUF[0] = 0x00;
453  EP0BCL = 0x02; /* 2 bytes */
454  break;
455 
456  case 0x82: // IN, Endpoint (Stall Bits)
457  switch (SETUPDAT[4] & 0xf) { /* extract number */
458  case 0: a=EP0CS; break;
459  case 1: a=(SETUPDAT[4]&0x80)?EP1INCS:EP1OUTCS; break;
460  case 2: a=EP2CS; break;
461  case 4: a=EP4CS; break;
462  case 6: a=EP6CS; break;
463  case 8: a=EP8CS; break;
464  default: a=1; break; /* or better Stall? */
465  }
466  EP0BUF[0] = a & 1; /* return stall bit or 1 in case of err */
467  EP0BCL = 0x02; /* 2 bytes */
468  break;
469  default: /* STALL indicating Request Error */
470  EP0CS = bmEPSTALL;
471  break;
472  }
