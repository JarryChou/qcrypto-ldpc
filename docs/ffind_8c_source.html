69 void findmax(int *buf1, int *buf2, double *maxval, double *sigma, double *mean,
70  int *pos, int size, int ecnt1, int ecnt2) {
71  int i; /* counter index */
72  double ai, ar, br, bi; /* temporary variables */
73  double sx, sxx;
74  int maxpos;
75 
76  /* clear double buffers / transfer int to double buffers */
77  ar = ((double)ecnt1) / size; /* to get mean at zero */
78  br = ((double)ecnt2) / size;
79  for (i = 0; i < size; i++) {
80  f1[i][0] = (double)buf1[i] - ar;
81  f1[i][1] = 0.;
82  f2[i][0] = (double)buf2[i] - br;
83  f2[i][1] = 0.;
84  }
85  /* do forward transformations */
86  fftw_execute(plan1);
87  fftw_execute(plan2);
88  /* do conjugate and multiplication into array 1 */
89  for (i = 0; i < size; i++) {
90  ar = f1[i][0];
91  ai = f1[i][1];
92  br = f2[i][0];
93  bi = f2[i][1];
94  f1[i][0] = ar * br + ai * bi;
95  f1[i][1] = ar * bi - ai * br;
96  }
97  /* do do backtransform */
98  fftw_execute(plan3);
99  /* evaluate max, stddev and mean */
100  *maxval = 0.;
101  maxpos = 0;
102  sxx = 0;
103  sx = 0.;
104  for (i = 0; i < size; i++) {
105  ar = f1[i][0];
106  if (ar > *maxval) {
107  maxpos = i;
108  *maxval = ar;
109  }
110  sx += ar;
111  sxx += (ar * ar);
112  }
113  /* return values properly back */
114  *mean = sx / size;
115  *sigma = sqrt(sxx / size - (*mean) * (*mean));
116  *pos = maxpos;
117 }

123 /* procedure to fill the fast and slow folded event time functions. Fh
124  contains the source file handle, the buffers are of length 2^order
125  and the targeted bin width is 2^fres and 2^s_res, in multiples
126  of 1/8 nsec */
127 void fill_periodicals(FILE *fh, int *buf_fast, int fres, int *buf_slow,
128  int sres, int order, int *ecnt) {
129  long long int intime;
130  long long int mask = (1 << (order)) - 1;
131  while (EOF != fscanf(fh, "%lld", &intime)) {
132  buf_fast[(int)(mask & (intime >> fres))]++;
133  buf_slow[(int)(mask & (intime >> sres))]++;
134  *ecnt += 1;
135  }
136 }
