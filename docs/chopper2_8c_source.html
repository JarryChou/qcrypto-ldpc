24 /* program to partition the output of the timestamp card on the comparator
25  side. Consumes the timestamp signal, and generates type-1 packets out of
26  it, either in a single file/pipe, or a directory. A logfile for process
27  control can be generated. Filespecs for input and type-1 stream can be
28  found in the filespec whitepaper.
29 
30  status: worked on feb7 2006 with >53k epochs
31  added -V 3 option for counting singles 21feb06chk not tested yet
32  fixed neg epoch overflow check 7.5.06chk
33  merged in changes for deviceindependent protocol 29.7.2009chk.

37  usage: chopper2 [-i infilename]
38  -O outfilename1 | -D outfiledir1 | -S socket1
39  [-l logfile ] [-V verbosity] [ -F ]
40  [-U | -L]
41  [-m maxtime ]
42  [-4 | -6]
43 
44  implemented options:
45 
46  DATA STREAM OPTIONS:
47  -i infilename: filename of source events. Can be a file or a socket
48  and has to supply binary data according to the raw data
49  spec from thetimestamp unit.
50  -O fname1: Outfile name for type 1 raw sifting time files.
51  This option saves all type 1 packets into a the file/FIFO
52  named fname1
53  -D dir1: All type-1 packets are saved into the directory dir1, with
54  the file name being the epoch (filling zero expanded)
55  in hex. Filename is not padded at end.
56  ENCODING OPTIONS:
57  -U: universal epoch; the epoch is not only derived from the timestamp
58  unit digits, but normalized to unix time origin. This needs the
59  timestamp unit to emit event data with an absolute time tag.
60  For this to work, the received data cannot be older than xxx hours,
61  or an unnoted ambiguity error will occur.
62  -L: local timestamps only. The epoch is calculated from the
63  unmodified timestamp info as received from the unit. This is the
64  default.
65 
66  LOGGING & NOTIFICATION
67  -l logfile: Each emitted epoch packet index is logged into this file. The
68  verbosity level controls the granularity of details logged.
69  if nothing is specified, STDOUT is used for logging.
70  -V level: Verbosity level control. level is integer, and by default set
71  to 0. The logging verbosity criteria are:
72  level<0 : no logging
73  0 : log only epoc number (in hex)
74  1 : log epoch, length without text
75  2 : log epoch, length with text
76  3 : log epoch and detailled event numbers for single
77  event counting. format: epoch and 5 cnts spc separated
78  -F : flushmode. If set, the logging output gets flushed after
79  every write attempt.
80  -4: full backwards compatibility option with logging where
81  single counts include local coincidences. Also reduces the
82  number of events in the output log to five (total cnts and
83  individual detector lines).
84  -d debugname specifies the file name of an optional debugging log file.
85  If this file is not provided, no debugging log file is
86  generated. The old debug filename was hardcoded as
87  /tmp/cryptostuff/choplog2 and has now been removed. Check
88  downstream processing if needed.
89 
90 PROTECTION OPTION
91  -m maxnum: maximum time for a consecutive event to be meaningful. If
92  the time difference to a previous event exceeds this time,
93  the event is discarded assuming it has to be an error in the
94  timing information. Default set to 0, which corresponds to
95  this option being switched off. Time units is in microseconds.

230 int open_epoch(unsigned int te) {
231  #ifdef DEBUG
232  if (debuglog) {
233  fprintf(debuglog, "open_epoch, timestamp %d\n", te);
234  fflush(debuglog);
235  }
236  #endif
237  unsigned long long aep, tim;
238  unsigned int aepoc = 0, finalepoc;
239  int overlay;
240 
241  /* determine current epoc from raw epoc */
242  if (uepoch) { /* take universal epoch */
243  tim = time(NULL);
244  aep = (tim * 1953125) >> 20; /* time in epocs */
245  aepoc = (unsigned long)(aep & 0xffffffff);
246  overlay = ((aepoc >> 15) & 3) | /* from absolute epoc */
247  ((te >> 13) & 0xc); /* from timestamp epoc */
248  finalepoc = (aepoc & 0xfffe0000) + te + overlay_correction[overlay];
249  #ifdef DEBUG
250  if ((debuglog) && overlay_correction[overlay]) {
251  fprintf(debuglog, "ovrly corr; tim: %lld, te: %08x, overlay: %08x\n", tim,
252  te, overlay);
253  fflush(debuglog);
254  }
255  #endif
256  } else {
257  finalepoc = te;
258  }
259  /* printf("finalepoch: %08x, aepo: %08x\n",finalepoc,aepoc); */
260  /* populate headers preliminary */
261  head1.tag = uepoch ? TYPE_1_TAG_U : TYPE_1_TAG;
262  head1.length = 0;
263  head1.epoc = finalepoc;
264  head1.bitsperentry = 49;
265  head1.basebits = 4; /* according to whitepaper */
266 
267  /* initialize output buffers and temp storage*/
268  index1 = 0;
269  return 0;
