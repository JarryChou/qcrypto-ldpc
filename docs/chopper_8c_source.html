24  program to digest the binary input from a timestamp card, separate
25  it into sections corresponding to one epoc, and emit packets for
26  sifting/coincidence purpose both for transmission over the classical
27  channel and for local storage for later key usage.

36  -O outfilename2 | -D filedir2 | -S socket2
37  -o outfilename3 | -d outfiledir3 | -s socket3
38  [-l logfile] [-F ] [-V verbosity]
39  [-U | -L]

45  implemented options:
46 
47  DATA STREAM OPTIONS:
48  -i infilename: filename of source events. Can be a file or a socket
49  and has to supply binary data according to the raw data
50  spec from thetimestamp unit.
51 
52  -O fname2: Outfile name for type 2 compressed sifting time files.
53  This option saves all type 2 packets into a the file/FIFO
54  named fname2
55  -D dir2: All type-2 packets are saved into the directory dir2, with
56  the file name being the epoch (filling zero expanded)
57  in hex. Filename is not padded at end.
58  -o fname3: same as option -O, but for type-3 files
59  -d dir3: same as option -d, but for type-3 files
60 
61  ENCODING OPTIONS:
62  -U: universal epoch; the epoch is not only derived from the timestamp
63  unit digits, but normalized to unix time origin. This needs the
64  timestamp unit to emit event data with an absolute time tag.
65  For this to work, the received data cannot be older than xxx hours,
66  or an unnoted ambiguity error will occur.
67  -L: local timestamps only. The epoch is calculated from the
68  unmodified timestamp info as received from the unit. This is the
69  default.
70  -q depth:Number of bits used to compress the timing data. default is 17,
71  which should be optimal for ... kevents/sec.
72  -Q num: filter time constant for bitlength optimizer. The larger the
73  num, the longer the memory of the filter. for num=0, no change will
74  take place. This is also the default.
75 
76  LOGGING & NOTIFICATION
77  -l logfile: Each emitted epoch packet index is logged into this file. The
78  verbosity level controls the granularity of details logged.
79  If no filename is specified, logging is sent to STDOUT.
80  -V level: Verbosity level control. level is integer, and by default set
81  to 0. The logging verbosity criteria are:
82  level<0 : no logging
83  0 : log only epoc number (in hex)
84  1 : log epoch, length without text
85  2 : log epoch, length with text
86  3 : log epoch, length, bitlength with text
87  4 : log epoch, detector histos without text
88  -4 sets the number of detectors to be logged to 4 (default)
89  in service mode. Has no effect in crypto modes
90  -6 sets the number of detectors to be logged to 6. This option
91  has only an effect in service mode
92  -F flushmode. If set, the logging info is flushed after every
93  write. useful if used for driving the transfer deamon.
94  -e debugfile choose a file for debug logging. This used to be hardcoded to
95  the file /tmp/cryptostuff/chopdebug, but is turned off for use
96  with other applications.
97  -W Write instead of append to logfile.
98 
99  PROTOCOL OPTION
100  -p num: select protocol option. defines what transmission protocol
101  is run by selecting what event bits are saved in which
102  stream. option 1 is default.
103  0: service protocol. both type-2 stream and type-3 stream
104  contain the raw detector information.
105  1: BB84 standard protocol. The type-2 stream contains one bit
106  of basis information, the type-3 stream one bit of
107  value information. The detector sequence is hardcoded in
108  the header.
109  2: rich BB84. As before, but two bits are transmitted. if the
110  msb is 0, the lsb has BB84 meaning, if msb is 1, a multi-
111  or no-coincidence event was recorded (lsb=1), or a pair
112  coincidence was detected (lsb=0).
113  ---------modifications for device-independent --------
114  3: six detectors connected to this side, used for the
115  device-independent mode. three transmitted bits, indicating
116  bell basis or key basis
117  4: four detectors connected to this side, device-indep
118  operation. only time is transmitted.
119  ---------modifications to BC protocol-----------------
120  5: Like 1, but no basis is transmitted, but basis/result
121  kept in local file
122  BASIS OPTION
123  -b alicebasis: A string of 4 characters comprising of 'H','V','-','+' (case sensitive) to
124  represent the order of bases for the base sequence on Alice's side
125  where '-b "V-H+" ' represents (lsb)V - H + (msb).
126  -B bobbasis: A string of 4 characters comprising of 'H','V','-','+' (case sensitive) to
127  represent the order of bases for the base sequence on Bob's side.
128  where '-B "V-H+" ' represents (lsb)V - H + (msb)
129 
130  1. By default both alicebasis & bobbasis are "V-H+".
131  2. It is assumed both sides use the same basis (V-H+), but you can use
132  this to customize alice's data to match bob's basis if needed.
133  3. As of now only chopper offers this customization.
134  4. Note that so long as the characters are unique, you can use any character to
135  represent a basis.
136 
137 PROTECTION OPTION
138  -m maxnum: maximum time for a consecutive event to be meaningful. If
139  the time difference to a previous event exceeds this time,
140  the event is discarded assuming it has to be an error in the
141  timing information. Default set to 0, which corresponds to
142  this option being switched off. Time units is in microseconds.

355 /* Rectifying the basis:
356  * where "V-H+" represents (lsb)V - H + (msb)
357  * Given two char arrays (basisSequences) of "V-H+" "HV-+" etc,
358  * we can precompute the bit indexes to swap (using XOR operations).
359  * This means a maximum of 3 swaps (last item will always be in proper position after 3 checks)
360  * With every swap, the current bit will be in the correct position.
361  *
362  * Swap index: iterating from the least significant bit (lsb). -1 if swap not needed, else store
363  * the relative position of the bit to swap with (will never be 0 or negative because a swapped bit
364  * will always be correct).
365  *
366  * The swap procedure is inlined for efficiency. Search for "t_state"
367 */
368 #define BASIS_LENGTH 4 /* number of basis */
369 #define BASIS_FORMAT "%4s" /* for sscanf of basis */
370 char basisSequences[2][BASIS_LENGTH + 1] = {"V-H+", "V-H+"}; // 0: Alice, 1: Bob
371 char basisRelativeSwapIndexes[BASIS_LENGTH - 1];
372 int basisSwapNeeded = 0;
373 int basisSwapXor = 0;
374 
375 #define DEBUG_BASIS 1
376 #undef DEBUG_BASIS
377 
378 /* Precompute swap indexes as described above.
379  * Precondition: basisSequences should contain basis char sequences.
380  * Postcondition: basisRelativeSwapIndexes populated. basisSequences will be consumed in the process.
381  * Returns 0 if success, emsg code if fail
382  */
383 int precomputeSwapIndexes() {
384  //Initialize basisRelativeSwapIndexes to all negative
385  for (size_t i = 0; i < BASIS_LENGTH - 1; i++) {
386  basisRelativeSwapIndexes[i] = -1;
387  }
388 
389  char tmp = 0;
390  if (strlen(basisSequences[0]) != 4 || strlen(basisSequences[1]) != 4) {
391  return -emsg(38);
392  }
393  for (size_t currIndex = 0; currIndex < 3; currIndex++) {
394  // If found mismatch
395  #ifdef DEBUG_BASIS
396  fprintf(stderr, "%c %c\n", basisSequences[0][currIndex], basisSequences[1][currIndex]);
397  #endif
398  if (basisSequences[0][currIndex] != basisSequences[1][currIndex]) {
399  basisSwapNeeded = 1;
400  // Search for position to swap
401  for (size_t targetIndex = currIndex + 1; targetIndex <= 3; targetIndex++) {
402  if (basisSequences[0][currIndex] == basisSequences[1][targetIndex]) {
403  // Store it once found
404  basisRelativeSwapIndexes[currIndex] = targetIndex - currIndex;
405  // Swap the sequences
406  tmp = basisSequences[0][currIndex];
407  basisSequences[0][currIndex] = basisSequences[1][targetIndex];
408  basisSequences[1][targetIndex] = tmp;
409  break;
410  }
411  }
412  }
413  }
414  #ifdef DEBUG_BASIS
415  fprintf(stderr, "%d %d %d\n", basisRelativeSwapIndexes[0], basisRelativeSwapIndexes[1], basisRelativeSwapIndexes[2]);
416  #endif
417  return 0;
418 }

484 /* opening routine to target files & epoch construction */
485 int open_epoch(unsigned int te) {
486  #ifdef DEBUG
487  if (debuglog) {
488  fprintf(debuglog, "open_epoch, timestamp %d\n", te);
489  fflush(debuglog);
490  }
491  #endif
492  unsigned long long aep, tim;
493  unsigned int aepoc, finalepoc;
494  int overlay;
495 
496  /* determine current epoc from raw epoc */
497  if (uepoch) { /* take universal epoch */
498  tim = time(NULL);
499  aep = (tim * 1953125) >> 20; /* time in epocs */
500  aepoc = (unsigned long)(aep & 0xffffffff);
501  overlay = ((aepoc >> 15) & 3) | /* from absolute epoc */
502  ((te >> 13) & 0xc); /* from timestamp epoc */
503  finalepoc = (aepoc & 0xfffe0000) + te + overlay_correction[overlay];
504  #ifdef DEBUG
505  if ((debuglog) && overlay_correction[overlay]) {
506  fprintf(debuglog, "ovrly corr; tim: %lld, te: %08x, overlay: %08x\n", tim,
507  te, overlay);
508  fflush(debuglog);
509  }
510  #endif
511  } else {
512  finalepoc = te;
513  }
514 
515  /* populate headers preliminary */
516  head3.tag = uepoch ? TYPE_3_TAG_U : TYPE_3_TAG;
517  head3.length = 0;
518  head3.epoc = finalepoc;
519  head3.bitsperentry = type3datawidth;
520 
521  head2.tag = uepoch ? TYPE_2_TAG_U : TYPE_2_TAG;
522  head2.length = 0;
523  head2.timeorder = type2bitwidth;
524  head2.basebits = type2datawidth;
525  head2.epoc = finalepoc;
526  head2.protocol = proto_index;

530  sendword2 = 0;
531  resbits2 = 32;
532  index3 = 0;
533  sendword3 = 0;
534  resbits3 = 32;
535  return 0;

547 #define TYPE2_ENDWORD 1 /* shortword to terminate a type 2 stream */
548 /* int log_correcttable[] =
549  {35, 37, 38, 39, 41, 42, 43, 44, 45, 46, 47, 47, 48, 49, 50, 51}; */
550 /* flush output buffers and submit files */
551 int close_epoch() {
552  char ffnam[FNAMELENGTH + 10];
553  int retval, i, optimal_width, j;
554  unsigned int average_distance, t1;
555 
556  if (!thisepoch_converted_entries) return 0; /* no data collected */
557 
558  /* finish stream 2 entries */
559  t1 = TYPE2_ENDWORD << type2datawidth; /* add closing word */
560  /* save timing and transmit bits */
561  if (resbits2 >= bitstosend2) {
562  sendword2 |= (t1 << (resbits2 - bitstosend2));
563  resbits2 = resbits2 - bitstosend2;
564  if (resbits2 == 0) {
565  outbuf2[index2++] = sendword2;
566  sendword2 = 0;
567  resbits2 = 32;
568  }
569  } else {
570  resbits2 = bitstosend2 - resbits2;
571  sendword2 |= (t1 >> resbits2);
572  outbuf2[index2++] = sendword2;
573  resbits2 = 32 - resbits2;
574  sendword2 = t1 << resbits2;
575  }
