3 # debugMode: 0 or 1. If 1, then pfind is run a second time to record the logs.

15  echo "Missing parameter: epoch"
16  echo "Usage: pfind_then_costream.sh debugMode epoch"

22 # Body
23 # ===

25 # Check file's existence on Bob's side

28  echo "BOB T1 doesn't exist."

33 # Ordered by T1 & T2
34 # That thing at the end simply converts hex to int
35 # Note: Using this method you cannot log while getting the time diff
36 timeDiff=$(BOB/EXE/PFIND\
37  -I BOB/DIR/T1/$epoch \
38  -i BOB/DIR/TCP_DEST/$epoch \
39  -e $((16#$epoch)) )
40

43  BOB/EXE/PFIND\
44  -I BOB/DIR/T1/$epoch \
45  -i BOB/DIR/TCP_DEST/$epoch \
46  -e $((16#$epoch)) \
47  -l BOB/DIR/LOGS/pfind/logs.txt \
48  -V 3
49 
50  # Helper prints
51  echo "time diff is ${timeDiff}"
52  printf "\n Running costream \n\n"
53 fi
54

56  echo "Something went wrong with pfind"

62 # Note that it does not do a clean exit if you pass in a filename, but it does generate output & logs

64 # WARNING: COSTREAM opens costream_tlog as its debuglog

77  -n BOB/DIR/LOGS/costream/general_logs.txt \
78  -W \

81  -q 1
