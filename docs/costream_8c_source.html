343 void FILL_DEC_PROTO0(int *t) { /* parameter is 8 bits wide , with the stream-2
344  bits in bit7..4 of p3, stream-1 bits in lsbits.
345  result is 8 bits for stream-3 (a copy of p3),
346  and the four bits in stream-1 to stream-4.
347  The decision bit (bit 12) is always 1. */
348  int p3;
349  for (p3 = 0; p3 < 256; p3++) { /* all patterns */
350  t[p3] = p3 | /* for stream-3 data */
351  ((p3 & 0xf) << 8) | /* for stream-4 data */
352  0x1000; /* decision bit */
353  }
354 }
