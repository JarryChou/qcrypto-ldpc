96 /* transmit testregister t, output divider n, main divider m bits to the

98 int _rfsource_set_registers(int handle, int t, int n, int m){
99  int u;
100  if (t & ~7) return -1; /* t range overflow */
101  if (n & ~3) return -1; /* output divider overflow */
102  if (m & ~0x1ff) return -1; /* main divider overflow */
103 
104  /* assemble data word and transmit the 14 bits to the usb device */
105  u=((t & 7)<<11) | ((n & 3)<<9) | (m & 0x1ff);
106  ioctl(handle, Send_RF_parameter, u); /* error treatment?? */
107  return 0;
108 }

110 /* optimize frequency assuming a reference freq. returns exact freq or <0
111  on error. All Frequencies are given in kilohertz. */
112 int adjust_rfsource(int handle, int ftarget, int fref){
113  int np; /* out_divider power; division ratio 2,4,8,1 for np=0,1,2,3 finally*/
114  int m; /* main divsion ratio */
115  int tmp;
116  if (fref<10000 || fref>20000) return -5; /* reference out of range */
117  if (ftarget <50000) return -3;/* frequency below VCO capability */
118  if (ftarget>800000) return -2; /* frequency exceeds VCO capability */
119 
120  /* calculate raw division power to keep VCO between 400 and 800 MHz */
121  tmp = (800000/ftarget);
122  if (tmp>16) return -3; /* frequency below VCO capability */
123  for (np=1;tmp>>np;np++); /* np = 1,2,3,4 for ratios 1,2,4,8*/
124 
125  /* calculate main divider setting */
126  m=(ftarget <<(np-1))/(fref>>3); /* should give right main divider ratio */
127  if (m<1 || m>0x1ff) return -4; /* main divider out of range */
128  /* send this to the chip, test mode switched off;
129  correction to np in the chip; ratio 2,4,8,1 for n_value=0,1,2,3 */
130  if (_rfsource_set_registers(handle, 0,(np+2)&3,m)) return -1; /* some err */
131  /* calculate generated frequency in kilohertz; true frequency may be off
132  up to a kHz due to rounding */
133  return ((1<<np)*fref*m)>>4;
134 }
