744  cascade_fixParityIntervals(pb, inh_idx);
745

747  pb->leakageBits += cscData->diffBlockCount; /* for incoming parity bits */

754  outMatch = &outParity[(wordCount(cscData->diffBlockCount))];

757  lostBitsCount = cscData->diffBlockCount;

760  thispass = (cscData->binarySearchDepth & RUNLEVEL_LEVELMASK) ? 1 : 0;

764  } else if (thispass == 1) {
765  d = pb->permuteBufPtr; // level 1
766  }

769

771  if (cscData->binarySearchDepth & RUNLEVEL_BICONF) {
772  biconfmark = 1;
773  d = pb->testedBitsMarker;

778  for (i = 0; i < cscData->diffBlockCount; i++) {
779  matchResult <<= 1;

783  lastBitIndex = cscData->diffidxe[i]; /* old bitindices */
784

787  lostBitsCount -= 2; /* No initial parity, no outgoing */

792  if (biconfmark) flipBit(d2, firstBitIndex);
793  flipBit(d, firstBitIndex);
794  pb->correctedErrors++;
795  lostBitsCount -= 2; /* No initial parity, no outgoing */
796  cscData->diffidx[i] = firstBitIndex + 1; /* mark as emty */
797  goto skipparity; /* no more parity evaluation, skip rest */
798  }
799 
800  midBitIndex = firstBitIndex + (lastBitIndex - firstBitIndex + 1) / 2 - 1; /* new lower mid bitidx */
801  tmpSingleLineParity = helper_singleLineParity(d, firstBitIndex, midBitIndex);
802  if (((inh_data[wordIndex(i)] & uint32AllZeroExceptAtN(i)) ? 1 : 0) == tmpSingleLineParity) {
803  /* same parity, take upper half */
804  firstBitIndex = midBitIndex + 1;
805  cscData->diffidx[i] = firstBitIndex; /* update first bit idx */
806  matchResult |= 1; /* indicate match with incoming parity */
807  } else {
808  lastBitIndex = midBitIndex;
809  cscData->diffidxe[i] = lastBitIndex; /* update last bit idx */
810  }
811  /* If this is end of interval, correct for error */
812  if (firstBitIndex == lastBitIndex) {
813  if (biconfmark) flipBit(d2, firstBitIndex);
814  flipBit(d, firstBitIndex);
815  pb->correctedErrors++;
816  lostBitsCount--; /* we don't reveal anything on this one anymore */
817  goto skipparity;
818  }
819  /* Else, prepare new parity bit */
820  midBitIndex = firstBitIndex + (lastBitIndex - firstBitIndex + 1) / 2 - 1; /* new lower mid bitidx */
821  parityResult |= helper_singleLineParity(d, firstBitIndex, midBitIndex); /* save parity */
822 
823  skipparity:

825  /* save stuff in outbuffers */
826  outMatch[wordIndex(i)] = matchResult;
827  outParity[wordIndex(i)] = parityResult;
828  }
829  }
830  /* Cleanup residual bit buffers */
831  if (modulo32(i) != 0) {
832  outMatch[wordIndex(i)] = matchResult << (32 - modulo32(i));
833  outParity[wordIndex(i)] = parityResult << (32 - modulo32(i));
834  }
835 
836  /* a blocklength k decides on a max number of rounds */
837  if ((cscData->binarySearchDepth & RUNLEVEL_ROUNDMASK) < log2Ceil((pb->processingState == PSTATE_DOING_BICONF)
838  ? (cscData->biconfLength) : (thispass ? cscData->k1 : cscData->k0))) {

840  pb->leakageBits += lostBitsCount;
841  comms_insertSendPacket((char *)outgoingMsgHead, outgoingMsgHead->base.totalLengthInBytes);
842  return 0;
843  }

846  nonsent parities */
847

849  fix_permutedbits(pb);
850

852  while (True) { /* just a break construction.... */
853  cscData->binarySearchDepth = thispass ? RUNLEVEL_FIRSTPASS : RUNLEVEL_SECONDPASS;
854  cscData->diffBlockCount = cascade_prepParityListAndDiffs(pb, 1 - thispass); /* new differences */
855  /* error if wrong pass */
856  if (cscData->diffBlockCount == -1)
857  return 74;
858 
859  // break if no more errors in block
860  if ((cscData->diffBlockCount == 0) && (thispass == 1))
861  break;
862 
863  // if more space needed
864  if (cscData->diffBlockCount > cscData->diffBlockCountMax) {
865  // Reassign the diff buffer
866  free2(cscData->diffidx);
867  cscData->diffBlockCountMax = cscData->diffBlockCount;

871  return 54;
872  // end of interval

874  }

877  return cascade_prepFirstBinSearchMsg(pb, 1 - thispass);
878  }
879

881 
882  /* check for biconf reply */
883  if (pb->processingState == PSTATE_DOING_BICONF) {

901  cscData->biconfRound = 0; /* first BICONF round */
902  return cascade_initiateBiconf(pb);
