26 
27  program to read in events from the timestamp card and do the first
28  processing to the data. Configuration happens by commandline options.
29 
30  version for usb card....first attempt 17.12.06chk
31  status: version as of 18.7.19 chk
32 
33  output is directed to stdout in a configurable form, and data acquisition
34  is controlled via some signals.
35 
36  usage: readevents [-t treshold | -T tresh_absolute] [-q maxevents]
37  [-r | -R ] [-a outmode] [other options, see below]
38 
39  command line options:
40  -t treshold :set input treshold to a given value in DAC units,
41  ranging from 0 to 4095. Defaults to 2905.
42  -T tresh_absolute :set input treshold to an absolute value, given in
43  millivolts
44  -q maxevents : quit after a number of maxevents detected events.
45  Default is 0, indicating eternal operation.
46  -r : starts immediate acquisition after card setup.
47  This is default.
48  -R : Initializes card, but does not start acquisition
49  until a SIGUSR1 is received.
50  -a outmode : Defines output mode. Defaults currently to 0. Currently
51  implemented output modes are:
52  0: raw event code patterns are delivered as 64 bit
53  hexadecimal patterns (i.e., 16 characters)
54  separated by newlines. Exact structure of
55  such an event: see card description.
56  1: event patterns with a consolidated timing
57  (i.e., corrected for spurious samplings from the
58  nudaq card)
59  are given out as 64 bit entities. Format:
60  most significant 49 bits contain timing info
61  in multiples of 0.125 nsec, least significant
62  4 bits contain the detector pattern.
63 
64  2: output consolidated timing info as 64 bit patterns
65  as in option 1, but as hext text.
66  3: output only the phase pattern (for tests only) as
67  decimal values from 0-511 from consolidated timing
68  info.
69  4: output as three space-separated hex patterns for
70  msl, lsl, and pattern
71  5: output as three space-separated decimal patterns for
72  msl, lsl, and pattern
73 
74  -v verbosity : selects how much noise is generated on nonstandard
75  events. All comments go to stderr. A value of 0
76  means no comments. Default is 0.
77  -s skew : set timing skew DAC to value skew (must be
78  between 0 and 4095); default: 2000 (old: 1535)
79  -j cal : run in calibrate mode and use calibration value cal.
80  -A : Output in absolute time. The timestamp mark is added to
81  the unix time evaluated upon starting of the timestamp
82  card; the resulting time is truncated to the least
83  significant 49 bit in multiples of time units (1/8 nsec).
84  -F : flush after each output line. useful for debugging.
85  -p phasepatt : select phase correction table. currently implemented:
86  -1: no phase correction
87  0: phase table for board revision 0
88  2: phase table calibrated for g2 measurement (default)
89  -e : choose external clock (assume 10 MHz)
90  -i : use internal 10 MHz clock. This is the default.
91  -x : exclude spurious outputs (larger than 8 average
92  differences )
93  -S skipnum : skip a given number of events at the beginning (to
94  avoid stale entries in the dma pipeline ?). Default is 0.
95  -m opt: marking option. if set to 1, the unused bits are used to
96  indicate time slips between timestamp time and CPU time.
97  For opt = 0, the phase info is given in bits 4..12
98  (default). for opt=2, all bits are zero.
99  -d s1,s2,s3,s4: add skew times to individual detectors. The supplied
100  values must be comma-separated and identify corrections
101  in multiples of 125 ps for each detector. This skew
102  is added to the timing information in case only a single
103  detector fires. Values can be positive or negative.
104  This option is only executed if the -A option is active
105  in order to minimize processing in the central timing
106  routine.
107  -D s1,s2,s3,s4,s5,s6,s7,s8 : Same as the -d option, but this time for
108  taking care of up to 8 detectors. Detector assignment:
109  line 1: det 1
110  line 2: det 2
111  line 3: det 3
112  line 4: det 4
113  line 1-2: det 5
114  line 2-3: det 6
115  line 3-4: det 7
116  line 4-1: det 8
117  -u usb flush mode is on. If no events were detected
118  during one periode, the flush option is activated
119 
120  -U devicename: allows to draw the raw data from the named device node.
121  If not specified, the default is /dev/ioboards/timestamp0
122  -Y y1,y2,y3,y4: ignores detector events if they fall within a certain time
123  of the last event seen by a particular detector. The dead times
124  y1...y4 are measured in multiples of 125ps.
125 
126  Signals:
127  SIGUSR1: enable data acquisition. This causes the inhibit flag
128  to be cleared, and data flow to start or to be resumed.
129  SIGUSR2: stop data acquisition. leads to a setting of the inhibit flag.
130  Data in the FIFO and the DMA buffer will still be processed.
131  SIGTERM: terminates the acquisition process in a graceful way (disabling
132  DMA, and inhibiting data acquisition into the hardware FIFO.
133  SIGPIPE: terminates the accquisition gracefully without further
134  messages on stderr.

422 /* function to digest data from the DMA buffer. quadsprocessed and quadsread
423  represent indices to 32-bit entities in the DMA buffer. return value
424  is the number of processed 32-bit-entries, or <0 if a DMA buffer overflow
425  occured.
426 */
427 int process_quads(void *sourcebuffer, int startquad, int endquad) {
428  unsigned int *events;
429  int startindex, endindex,i;
430  int numberofquads;
431  int j; /* processing variables */
432  unsigned int ju; /* for parsing binary numbers */
433  unsigned int u; /* contains events */
434  /* main processing variables */
435  int quadsthere;
436  unsigned int cv, cvd, v1, dv, fastcnt, b0;
437  char *formatstring;
438  static char formatstring_1[] = "event: msl: %08x; lsl: %08x\n";
439  static char formatstring_2[] = "%08x%08x\n";
440  int markit=0; /* for debugging time error */
441  unsigned long long current_time;
442  int pattern; /* holds detector pattern */
443 
444  events = (unsigned int *)sourcebuffer;
445  numberofquads = (endquad - startquad) & QUADMASK3 ; /* anticipate roll over*/
446 
447  /* what if startquad == endquad? */
448  if (numberofquads == 0) return 0; /* only look for pos transfers */
449  /* complain if buffer is too filled */
450  if ( numberofquads > ((int) dmabuf_complainwatermark) ) {
451  fprintf(stderr,"numofquads: %d, complainwm: %d\n",numberofquads, ((int) dmabuf_complainwatermark));
452  return -1;
453  }
454 
455  startindex = startquad % dmasize_in_longints;
456  endindex = endquad % dmasize_in_longints;
457 
458  switch (outmode) {
459  case 0: /* just for simple printout */
460  for (i=startindex;i!=endindex;i=((i+1) % dmasize_in_longints)) {
461  u=events[i];
462  if (verbosity) { /* long version */
463  printf("index: %04d, value: %08x :",i,u);
464  for (ju=0x80000000;ju;ju>>=1) printf("%d",((ju&u)?1:0));
465  printf("\n");
466  } else {
467  printf("%08x\n",u); /* only hex code */
468  }
469  }
470  /* check if max event mechanism is active */
471  if (maxevents) {
472  currentevents++;
473  if (currentevents==maxevents) {
474  terminateflag=1;
475  return numberofquads;
476  }
477  }
478  return numberofquads;
479 
480  /* do first processing in other cases */
481  case 1:case 2:
482  i=startindex;
483  j=0; /* start target index */
484  for (quadsthere=numberofquads;
485  quadsthere>1;
486  quadsthere-=2,i=((i+2) % dmasize_in_longints)) {
487  /* extract coarse val and check for consistency */
488  cv=events[i+1];
489  dv=events[i];
490  /* check for dummies - to be implemented ??*/
491  if (!(cv | dv)) { /* both are zero...indicates an error */
492  /* Commented out because of noise level
493  fprintf(stderr,"err: double zero\n"); */
494  continue;
495  }
496 
497  cvd=(cv>>16)-controltime_coarse+2; /* difference plus 2 */
498  /* for debugging stop */
499  if (cvd>4) {/* allow for approx 2 sec difference */
500  fprintf(stderr, "timing out of range; cv=%d, control=%d, dv=%d, idx: %d\n",cv,controltime_coarse,events[i],i);
501  /* continue; */
502  if (markoption==1) markit+=0x10;
503  /* FIXME: for debug: try not to realign 4-byte entities */
504  /* quadsthere--;i=((i+1) % dmasize_in_longints); */
505  continue;
506  }
507  /* now we should be consistent. no mixing necessry anymore */
508  /* get first and second entry */
509  v1=((dv & 0xc000)>>12) | ((dv & 0x30000)>>16); /* event lines */
510 
511  fastcnt=(dv & 0x3e00)<<10;
512  if (markoption==0)
513  markit= (dv<<4); /* bring phase pattern in place */
514  /* construct lower significant u32, containing c0-c12, c_1..c_4,
515  and the event bits in the least significant nibble.
516  order there: bit 3..0=inA..InD */
517  dv=(dv & 0xff000000) | /* bits c5..c12 */
518  fastcnt | /* bits c0..c4 */
519  phasetable[dv & 0x1ff] | /* do phase interpolation */
520  v1 | /* event lines */
521  (markit & 0x1ff0); /* for debugging */
522 
523  /* repair pipelining bug */
524  if ( (fastcnt < 0x00880000)) {
525  b0=dv & 0x80000000; /* remember carry */
526  dv+=0x01000000;
527  if (b0 && !(dv & 0x80000000)) cv++; /* eventually do carry */
528  }
529 
530  if(timemode==1) {
531  pattern=dv&0xf; /* detector pattern */
532  current_time = (((unsigned long long)cv) << 32)
533  + (unsigned long long)dv
534  /* correction for time skew of individual detectors */
535  + dayoffset[pattern];
536  if (!deadtimecorrect) {
537  outbuf[j].cv=(unsigned int) (current_time >> 32);
538  outbuf[j].dv=(unsigned int) (current_time & 0xffffffff);
539  } else {
540  if (current_time-lasttime[pattern] > ddeadpatt[pattern]) {
541  lasttime[pattern]=current_time;
542  outbuf[j].cv=(unsigned int) (current_time >> 32);
543  outbuf[j].dv=(unsigned int) (current_time & 0xffffffff);
544  } else {
545  lasttime[pattern]=current_time;
546  continue; /* next element in for loop */
547  }
548  }
549  } else {
550  outbuf[j].cv=cv; outbuf[j].dv=dv;
551  }
552  /* keep track of movin difference */
553  if (controltime_getmeone) {
554  controltime_cv=cv; controltime_dv=dv;controltime_getmeone=0;
555  }
556
